    # 2. 数组

    1.普通数组 只能用数字作为元素的名称
2.关联数组 可以用数字以及字符串作为元素的名称

语法格式: 
	数组名称[元素名称]=元素的值
	数组名称[索引]=索引的值
	数组名称[下标]=下标的值
	变量只能一个变量对应一个值
	数组一个变量中可以放多个值

    完善jumpserver跳板机
1.运维和开发属于不同的账号
2.运维和开发登录跳板机输入密码
3.开发的菜单和运维的菜单不同
    主菜单
        1.运维
        2.开发
    子菜单
        显示运维可以连接的服务器信息
        显示开发可以连接的服务器信息
[root@web01:scripts]#cat 1.sh 
#!/bin/bash
WEB02=10.0.0.8
NFS=10.0.0.31
while true
do
fun1(){
    echo -e "\t\t\t\t\t\033[34m1.运维\033[0m"
    echo -e "\t\t\t\t\t\033[34m2.开发\033[0m"
}
fun1

yw(){
    echo -e "\t\t\t\t\t\033[34m1.$WEB02\033[0m"
    echo -e "\t\t\t\t\t\033[34m2.$NFS\033[0m"
    echo -e "\t\t\t\t\t\033[34m3.显示子菜单\033[0m"
    echo -e "\t\t\t\t\t\033[34m4.返回上一级\033[0m"
}

dev(){
    echo -e "\t\t\t\t\t\033[34m1.$NFS\033[0m"
    echo -e "\t\t\t\t\t\033[34m2.显示主菜单\033[0m"
    echo -e "\t\t\t\t\t\033[34m3.返回上一级\033[0m"
}

read -p "选择你的角色编号: " num1

case $num1 in
  1)
    yw
    while true
    do
      read -p "请输入连接主机的编号3菜单:" num2
      if [ $num2 -eq 1 ];then
        ssh $WEB02
      elif [ $num2 -eq 2 ];then
        ssh $NFS
      elif [ $num2 -eq 3 ];then
        clear
        yw
      elif [ $num2 -eq 4 ];then
        break
      fi
    done
    ;;
  2)
    dev
  ;;
  *)
    echo "Usage: $0[1|2选择角色]"
esac
done
