    # 1.表达式之文件判断

    语法结构:
	语法1: test -f /etc/hosts # 判断hosts文件是否存在   了解
	语法2: [ -f /etc/hosts ] # 判断hosts文件是否存在  掌握
	
-f # 判断文件是否存在
-d # 判断目录是否存在
-e # 存在即为真
-r # 可读为真
-w # 可写真为
-x # 可执行为真

    # 2.条件表达式字符串比对

    语法结构: 条件表达式中都可以使用命令或者变量

	[ 字符串1 != 字符串2 ] # 不等于则为真
	[ $USER = root ]
	[ 5 -eq `cat passwd|wc -l` ]

    # 3.整数比较

    语法结构:
    [ 整数1 比较符 整数2 ]

-eq # 相等
-ne # 不等于
-gt # 大于

-lt # 小于
-le # 小于等于

    案例1.判断两个数字的大小
[root@web01:scripts]#cat 1.sh 
#!/bin/bash
expr $1 + $2 &>/dev/null
if [ $? -eq 0 ];then
  if [ $1 -gt $2 ]
  then
    echo "$1 > $2"
  elif [ $1 -lt $2 ]
  then
    echo "$1 < $2"
  else
    echo "$1 = $2"
  fi
else
  echo "必须传入数字"
  exit
fi

    案例3.case写一个nginx启动脚本

nginx启动方式两种:
    一种是systemctl
    一种是 /usr/sbin/nginx
    
玩一下，另一种启动Nginx的方法：
[root@web01:~]#/usr/sbin/nginx     启动nginx
[root@web01:~]#/usr/sbin/nginx -s stop  停止nginx
[root@web01:~]#/usr/sbin/nginx     启动nginx
[root@web01:~]#/usr/sbin/nginx -s stop && sleep 1 && /usr/sbin/nginx  重启nginx
[root@web01:~]#/usr/sbin/nginx -s reload  重载

写一个启动nginx的启动脚本：
[root@web01:~]#cat 1.sh 
#!/bin/bash
case $1 in
  start)
    /usr/sbin/nginx
    ;;
  stop)
    /usr/sbin/nginx -s stop
    ;;
  reload)
    /usr/sbin/nginx -s reload 
    ;;
  restart)
    /usr/sbin/nginx -s stop && sleep 1 && /usr/sbin/nginx
    ;;
  status)
    netstat -tnulp|grep nginx|awk '{print $4}'
    ;;
  *)
    echo "Usage $0 [start|stop|reload|restart|status]"
esac
